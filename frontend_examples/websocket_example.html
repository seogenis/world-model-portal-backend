Code Example

Here's how to connect to the WebSocket and handle updates:

// Initialize WebSocket connection
const socket = new WebSocket(`ws://your-host/ws/video/${jobId}`);

// Connection opened
socket.onopen = function(event) {
    console.log('WebSocket connection established');
};

// Listen for messages
socket.onmessage = function(event) {
    const update = JSON.parse(event.data);
    console.log('Received update:', update);
    
    // Update UI based on status
    updateStatus(update.status, update.message);
    updateProgress(update.progress);
    
    // If video is complete, show it
    if (update.status === 'complete' && update.video_url) {
        displayVideo(update.video_url);
    }
};

// Connection closed
socket.onclose = function(event) {
    if (!event.wasClean) {
        console.error('WebSocket connection closed unexpectedly');
    }
};

// Connection error
socket.onerror = function(error) {
    console.error(`WebSocket error: ${error.message}`);
};
